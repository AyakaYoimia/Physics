\chapter{算符的性质}

多项式$F(x)$，
\begin{gather*}
   [\hat{Q},F(\hat{P})]=\i \hbar F'(\hat{P})
   \\
   [\hat{P},F(\hat{Q})]=-\i \hbar F'(\hat{Q})
\end{gather*}

对于一对共轭坐标，
\begin{gather*}
   [\hat{Q},\hat{P}]=\i \hbar
   \\
   \intertext{定义}
   \hat{S}(\lambda)=\e^{-\frac{\i}{\hbar} \lambda \hat{P}},\quad \lambda \in \mathbb{R}
   \\
   \hat{S}^{\dagger}(\lambda)=\hat{S}^{-1}(\lambda)=\hat{S}(-\lambda)
   \\
   \hat{Q}\hat{S}(\lambda)=S(\lambda)(\hat{Q}+\lambda)
   \\
   \intertext{若}
   \hat{Q}\ket{q}=q\ket{q}
   \\
   \intertext{则}
   \hat{Q}\hat{S}(\lambda)\ket{q}=(q+\lambda)\hat{S}(\lambda)\ket{q}
\end{gather*}
$\hat{Q}$的全体特征值具有相同简并度
